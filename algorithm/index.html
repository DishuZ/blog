                      # 快速排序模板

                      void quick_sort(int q[], int l, int r)
{
    if (l >= r) return;
    
    int x = q[l + r >> 1], i = l - 1, j = r + 1;
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    
    quick_sort(q, l, j);
    quick_sort(q, j + 1, r);
}

                      ❓ 提问：

                      为什么在 [1] [2] 处修改为 q[i] <= x q[j] >= x 就会出现 Memory Limit Exceeded 的错误？

                      ✏️ 解答：

                      假如我们输入：1 1 1 1 1 1，若 [1] 处为 q[i] <= x，那么会在这里一直执行 while 循环，直到 i 下标越界，这里就出现报错 Memory Limit Exceeded 。

                      void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;
    
    int mid = l + r >> 1;
    merge_sort(q, l, mid), merge_sort(q, mid + 1, r);
    
    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
        if (q[i] <= q[j]) tmp[k ++ ] = q[i ++ ];
        else tmp[k ++ ] = q[j ++ ];
    while (i <= mid) tmp[k ++ ] = q[i ++ ];
    while (j <= r) tmp[k ++ ] = q[j ++ ];
    
    for (i = l, j = 0; i <= r; i ++, j ++ ) q[i] = tmp[j];
}

// 区间[l, r]被划分成[l, mid]和[mid + 1, r]时使用：
int bsearch_1(int l, int r)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;    // check()判断mid是否满足性质
        else l = mid + 1;
    }
    return l;
}
// 区间[l, r]被划分成[l, mid - 1]和[mid, r]时使用：
int bsearch_2(int l, int r)
{
    while (l < r)
    {
        int mid = l + r + 1 >> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}

                      • 数的范围 (opens new window)

                      本质：

                      • 不是单调性
                      • 而是数据满足某种性质：对于x ，在 q[mid] 左侧为 true，在 q[mid] 右侧为 false

                      二分一定有解。但题目不一定有解

                      一共三个要点：

                      1. 判断 mid 点在哪一边，包不包含在内；

                      2. mid 怎么取的问题：拿长度为 1 的区间 [r-1, r] 来分析，我们肯定是希望二分之后的结果是 [r-1, r-1] 或者是 [r, r] ，能够把区间夹成空，得到退出循环的条件。

                        两种模版二分的结果分别是 [l, mid],[mid+1, r] 和 [l,mid-1],[mid, r]。

                        • 前者 [l,mid-1],[mid, r] 中，如果让 mid = l + r >> 1 是向下取整，那么在 [r-1, r] 的区间中，mid = r - 1，把值代入 [mid, r] 中得到的是 [r-1, r] ，原区间一动不动，不是我们想要的结果。
                        • 同理，后者 [l, mid],[mid+1, r] 中，mid = l + r + 1 >> 1 的向上取整，那么在 [r-1, r] 的区间中，mid = r，l = r - 1 把值代入 [l, mid] 中得到 [r-1, r] ，原区间一动不动，不是我们想要的结果。

                      # 浮点型二分算法模型

                      bool check(double x) {/* ... */} // 检查x是否满足某种性质

double bsearch_3(double l, double r)
{
    const double eps = 1e-6;   // eps 表示精度，取决于题目对精度的要求
    while (r - l > eps)
    {
        double mid = (l + r) / 2;
        if (check(mid)) r = mid;
        else l = mid;
    }
    return l;
}

                      vector<int> add(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    
    int t = 0;
    for (int i = 0; i < A.size() || i < B.size(); i ++)
    {
        if (i < A.size()) t += A[i]; 
        if (i < B.size()) t += B[i];
        C.push_back(t % 10);
        t /= 10;
    }
    
    if (t) C.push_back(1);
    
    return C;
}

                      # 高精度减法

                      // 前提： A > B
vector<int> sub(vector<int> &A, vector<int> &B)
{
    vector<int> C;
    
    int t = 0;
    for (int i = 0; i < A.size(); i ++)
    {
        t = A[i] - t;
        if (i < B.size()) t -= B[i];
        C.push_back((t + 10) % 10);
        if (t < 0) t = 1;
        else t = 0;
    }
    
    while (C.size() > 1 && C.back() == 0) C.pop_back(); // 去掉前导 0
    
    return C;
}

// 判断是否有 A >= B
bool cmp(vector<int> &A, vector<int> &B)
{
    if (A.size() != B.size()) return A.size() > B.size();
    for (int i = A.size() - 1; i >= 0; i -- )
        if (A[i] != B[i]) return A[i] > B[i];
    return true;
}

                      # 高精度乘法

                      vector<int> mul (vector<int> &A, int b)
{
    vector<int> C;
    
    int t = 0;  // 进位
    for (int i = 0; i < A.size() || t; i ++ )
    {
        if(i < A.size()) t += A[i] * b;
        C.push_back(t % 10);
        t /= 10;
    }
    // 去掉前导 0
    while (C.size() > 1 && C.back() == 0) C.pop_back();
    
    return C;
}

                      // A,b 为非负整数，r为余数
vector<int> div (vector<int> &A, int b, int &r)
{
    vector<int> C;
    
    int t = 0;
    for (int i = A.size() - 1; i >= 0; i -- )
    {
        t = t * 10 + A[i];
        C.push_back( t / b );
        t %= b;
    }
    r = t;
    
    reverse(C.begin(), C.end());    // #include <algorithm>
    
    while (C.back() == 0 && C.size() > 1) C.pop_back();
    
    return C;
}

                      # 二维差分

                      void insert (int x1, int y1, int x2, int y2, int c)
{
    b[x1][y1] += c;
    b[x2 + 1][y1] -= c;
    b[x1][y2 + 1] -= c;
    b[x2 + 1][y2 + 1] += c;
